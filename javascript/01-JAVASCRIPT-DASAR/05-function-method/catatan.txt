// function benilai balik function
function hasilNilaiFunction (){
    const rand = Math.random();
    console.log(rand);
    if(rand > 0.3){
        const result = () => 'OK';
        return result;
       
    } else {
       
        const result = () => 'NOT OK';
        return result;
      
    }
}

const hasil = hasilNilaiFunction();
console.log(hasil);
